## Usage

export const importCode = `import {
  DropdownMenu,
  DropdownMenuContent,
  DropdownMenuItem,
  DropdownMenuLabel,
  DropdownMenuSeparator,
  DropdownMenuTrigger,
} from "@una/ui/dropdown-menu";`

export const usageCode = `<DropdownMenu>
  <DropdownMenuTrigger>打开</DropdownMenuTrigger>
  <DropdownMenuContent>
    <DropdownMenuLabel>我的账户</DropdownMenuLabel>
    <DropdownMenuSeparator />
    <DropdownMenuItem>个人资料</DropdownMenuItem>
    <DropdownMenuItem>账单</DropdownMenuItem>
    <DropdownMenuItem>团队</DropdownMenuItem>
    <DropdownMenuItem>订阅</DropdownMenuItem>
  </DropdownMenuContent>
</DropdownMenu>`
