## Usage

  Dialog,
  DialogContent,
  DialogDescription,
  DialogHeader,
  DialogTitle,
  DialogTrigger,
} from "@una/ui/dialog";`

  <DialogTrigger>打开</DialogTrigger>
  <DialogContent>

      </DialogDescription>
    </DialogHeader>
  </DialogContent>
</Dialog>`
