## Usage

export const importCode = `import {
  Pagination,
  PaginationContent,
  PaginationEllipsis,
  PaginationItem,
  PaginationLink,
  PaginationNext,
  PaginationPrevious,
} from "@una/ui/pagination";`

export const usageCode = `<Pagination>
